\begin{problem*}
	For $x,r\in\RR$ find
	\[
		\sum_{k=0}^n r^k\cos kx \text{ and } \sum_{k=0}^n r^k\sin kx.
	\]
	Then, assuming $|r|<1$ take their limits as $n\rightarrow\infty$.
\end{problem*}
